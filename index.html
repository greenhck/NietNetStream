const searchBar = document.querySelector('.search-bar'); searchBar.addEventListener('input', (event) => { const searchTerm = event.target.value.toLowerCase(); if (nStreamPlayerSection.classList.contains('flex')) { return; } let channelsToDisplay; if (searchTerm === '') { // 🟢 जब इनपुट खाली हो, तो हमेशा मास्टर लिस्ट (पूरा डेटा) दिखाएँ channelsToDisplay = window.masterChannelList; } else { // 🟢 फ़िल्टरिंग भी मास्टर लिस्ट पर ही करें channelsToDisplay = window.masterChannelList.filter(channel => channel.name.toLowerCase().includes(searchTerm) ); } // UI को अपडेट करें displayChannels(channelsToDisplay); });
